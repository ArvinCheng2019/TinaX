XCore.VFS = XCore.VFS or {}
XCore.Asset = XCore.VFS

---加载资源
XCore.Asset.LoadAsset = function(path,type)
    return CS.TinaX.VFS.I:LoadAsset(path,type)
end

---加载资源（异步）
XCore.Asset.LoadAssetAsync = function(path,type,callback)
    CS.TinaX.VFS.I:LoadAssetAsync(path,type,callback)
end


---移除资源引用（计数器： -1s, 太暴力啦）
function XCore.Asset.RemoveUse(path)
    CS.TinaX.VFS.I:RemoveUse(path);
end


---GC 
function XCore.Asset.GC()
    CS.TinaX.VFS.I:GC();
end
