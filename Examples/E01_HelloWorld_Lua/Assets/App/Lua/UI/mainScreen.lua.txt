-- luabehaviour MainScreen

local open_ui_param_mask;

function Awake()
    open_ui_param_mask= CS.TinaX.UIKit.OpenUIParam();
    open_ui_param_mask.UseMask = true;
    open_ui_param_mask.CloseByMask = true;
end

function Start()

    btn_1.onClick:AddListener(function()
        App.UIKit:OpenUIWithParamAsync("msgBox", open_ui_param_mask,function(uiEntity,err)
            if(err ~= nil) then
                printE("Open UI \"msgBox\" failed: " .. err.Message);
            else
                print("o((>ω< ))o");
            end
        end, "Hello,World", "Meow");
    end)

    btn_counter.onClick:AddListener(function()
        App.UIKit:OpenUIWithParamAsync("counter", open_ui_param_mask, function(uiEntity,err)
            if(err ~= nil) then
                printE("Open UI \"counter\" failed: " .. err.Message);
            else
                print("o((>ω< ))o");
            end
        end);
    end)

    btn_fetchData.onClick:AddListener(function()
        App.UIKit:OpenUIAsync("fetchData",function(uientity,err)
            --
        end);
    end)
end
